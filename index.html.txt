                <button class="btn" onclick="loadExternalSubtitle()">Naloži podnapis</button>

    <script>
        const items = document.querySelectorAll('.video-item');
        const player = document.getElementById('tvPlayer');
        const track = document.getElementById('playerTrack');
        let currentIndex = 0;

        // 1. Podpora za več formatov: Televizor bo samodejno predvajal formate (MP4, MKV, WebM), če jih podpira strojna oprema TV-ja.
        function changeVideo(index) {
            items.forEach(i => i.classList.remove('active'));
            items[index].classList.add('active');
            
            const url = items[index].getAttribute('data-url');
            player.src = url;
            player.load();
            player.play();
        }

        // 2. Iskanje na Podnapisi.org
        function openPodnapisi() {
            const query = document.getElementById('searchQuery').value;
            if(query) {
                // Odpre iskanje neposredno na spletni strani slo-podnapisi.net v novem zavihku brskalnika
                const searchUrl = `https://slo-podnapisi.net{encodeURIComponent(query)}`;
                window.open(searchUrl, '_blank');
            } else {
                alert("Prosimo, vpišite ime filma.");
            }
        }

        // 3. Ročno dodajanje podnapisov (Format .VTT)
        function loadExternalSubtitle() {
            const subUrl = document.getElementById('subUrlInput').value;
            if (subUrl) {
                track.src = subUrl; // Vstavi zunanji podnapis v predvajalnik
                alert("Podnapis uspešno naložen!");
            } else {
                alert("Vnesite veljaven URL naslov podnapisa.");
            }
        }

        // Upravljanje s tipkami (Gor/Dol/Enter) za Hisense daljinec
        document.addEventListener('keydown', function(e) {
            if (document.activeElement.tagName === 'INPUT') return; // Če tipkamo v iskalnik, onemogoči premikanje videa

            if (e.key === 'ArrowUp' && currentIndex > 0) {
                currentIndex--;
                changeVideo(currentIndex);
            } else if (e.key === 'ArrowDown' && currentIndex < items.length - 1) {
                currentIndex++;
                changeVideo(currentIndex);
            } else if (e.key === 'Enter') {
                if (player.requestFullscreen) player.requestFullscreen();
            }
        });

        // Klikanje z miško/kurzorjem
        items.forEach((item, index) => {
            item.addEventListener('click', () => {
                currentIndex = index;
                changeVideo(index);
            });
        });
    </script>
